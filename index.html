var box = document.querySelector(“#box”);
//事件监听 绑定事件 可以给一个对象绑定多个同样的事件
box.addEventListener(“click”,function(){},false); //冒泡 默认值是false
function boxClick(){
box.removeEventListener(“click”,boxClick); //移除事件
}
box.addEventListener(“click”,boxClick,true); //捕获

box.attachEvent("onmouseover",function(){});        //ie 6-10支持
function boxMouseover(){
    box.detachEvent("onmouseover",boxMouseover)       //移除事件
}
box.attachEvent("onmouseover",boxMouseover)



box.onclick = function(){          //绑定事件
    box.onclick = null;                //移除事件
}

box.onclick = function(event){
    event = event || window.event;
    event.target         //真正出发的那个元素   兼容性问题
    event.srcElement              //真正出发的那个元素
    var target = event.target || event.srcElement;
    event.currentTarget;         // 和 this 是一样的 指的是所绑定的那个对象
    event.eventPhase;            //获取事件阶段的3个值 1 捕获 2 目标  3 冒泡
    event.clientX;           //获取鼠标在浏览器可视区域的横坐标  比如:固定定位
    event.clientY        //获取鼠标在浏览器可视区域的纵坐标
    event.pageX;       //获取鼠标在页面上横坐标  包括 滚动出去的距离   比如:绝对定位
    event.pageY;          //获取鼠标在页面上纵坐标
    document.body.scrollLeft         //获取页面滚动出去距离 left距离
    document.body.scrollTop//获取页面滚动出去距离 top距离
    document.documentElement.scrollLeft       //获取页面滚动出去距离 left距离
    document.documentElement.scrollTop        //获取页面滚动出去距离 top距离
    var pageY = document.body.scrollTop + event.clientY;

    box.offsetLeft       //盒子距离最近带有定位的父元素的左侧距离
    box.offsetTop       //盒子距离最近带有定位的父元素的上侧距离

}

  2. git config –global user.name “自己的用户名”
    git config –global user.email “自己的邮箱”
    git config –list 查看git用户配置信息

  3. git add ./ 添加文件到暂存区门口
    git status 查看git提交的状态
    git commit -m ‘提交描述’ 将门口的文件拿过来放到暂存区

  4. 新建忽略文件 .gitignore 先配置忽略文件 在添加

  5. git log 查看提交记录
    git log –oneline 查看简洁版提交日志

  6. 版本回退
    git log 查看获取版本号
    git reset –hard 版本号 回退到对应的版本

  7. git branch 分支名字 创建分支
    git branch 获取分支信息
    git checkout 分支名字 切换分支

    git merge 分支名字 和并分支

  1. 将本地仓库代码上传到远程代码仓库

git remote add origin https://gitee.com/klxin/web22.git
git push -u origin master

  1. 下拉 上传
    git push 推送到远程仓库
    git pull 从远程仓库下拉

  2. 克隆
    git clone 仓库地址 （注意: 从远程仓库克隆到本地 已经关联好了）

  3. 实际操作
    11.1 现在远程创建一个仓库
